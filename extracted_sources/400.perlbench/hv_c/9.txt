SV**
Perl_hv_fetch(pTHX_ HV *hv, const char *key, I32 klen_i32, I32 lval)
{
    HE *hek;
    STRLEN klen;
    int flags;

    if (klen_i32 < 0) {
	klen = -klen_i32;
	flags = HVhek_UTF8;
    } else {
	klen = klen_i32;
	flags = 0;
    }
    hek = hv_fetch_common (hv, NULL, key, klen, flags,
			   HV_FETCH_JUST_SV | (lval ? HV_FETCH_LVALUE : 0),
			   Nullsv, 0);
    return hek ? &HeVAL(hek) : NULL;
}
