void set_last_dquant()
{
  Macroblock *currMB = &img->mb_data[img->current_mb_nr];

  if ((IS_INTERMV (currMB)  || IS_INTRA (currMB)  ) ||
    ((img->type==B_SLICE)     && currMB->cbp != 0)  )
  {
    // non-skip
    last_dquant = currMB->delta_qp;
  }
  else
  {
    // skip
    last_dquant = 0;
  }
}
