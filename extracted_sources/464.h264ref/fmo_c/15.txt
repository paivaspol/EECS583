int FmoGetPreviousMBNr (int CurrentMbNr)
{
  
  int  SliceGroupID = FmoMB2SliceGroup (CurrentMbNr); 
  CurrentMbNr--;
  while (CurrentMbNr>=0 &&  MBAmap[CurrentMbNr] != SliceGroupID)
    CurrentMbNr--;
  
  if (CurrentMbNr < 0)
    return -1;    // No previous MB in this slice
  else
    return CurrentMbNr;
}
