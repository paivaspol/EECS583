void free_mem_mincost (int***** mv)
{
  int i, j, k, l;

  for (i=0; i<img->width/4; i++)
  {
    for (j=0; j<img->height/4; j++)
    {
      for (k=0; k<img->max_num_references; k++)
      {
        for (l=0; l<9; l++)
          free (mv[i][j][k][l]);
        free (mv[i][j][k]);
      }
      free (mv[i][j]);
    }
    free (mv[i]);
  }
  free (mv);
}
