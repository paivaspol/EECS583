bool_t
xdr_bool (XDR *xdrs, int *bp)
{
#define XDR_FALSE	((long) 0)
#define XDR_TRUE	((long) 1)
  long lb;

  switch (xdrs->x_op)
    {
    case XDR_ENCODE:
      lb = *bp ? XDR_TRUE : XDR_FALSE;
      return xdr_putlong (xdrs, &lb);

    case XDR_DECODE:
      if (!xdr_getlong (xdrs, &lb))
	{
	  return FALSE;
	}
      *bp = (lb == XDR_FALSE) ? FALSE : TRUE;
      return TRUE;

    case XDR_FREE:
      return TRUE;
    }
  return FALSE;
#undef XDR_FALSE
#undef XDR_TRUE
}
