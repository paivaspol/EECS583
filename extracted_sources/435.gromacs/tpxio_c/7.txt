static void do_atom(t_atom *atom,int ngrp,bool bRead)
{
  do_real (atom->m);
  do_real (atom->q);
  do_real (atom->mB);
  do_real (atom->qB);
  do_ushort(atom->type);
  do_ushort(atom->typeB);
  do_int (atom->ptype);
  do_int (atom->resnr);
  if (file_version < 23) {
    ngrp = 8;
    atom->grpnr[8] = 0;
  }
  do_nuchar(atom->grpnr,ngrp);
}
