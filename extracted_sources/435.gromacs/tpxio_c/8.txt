static void do_grps(int ngrp,t_grps grps[],bool bRead)
{
  int i,j;
  bool bDum=TRUE;
  
  if (file_version < 23) {
    ngrp = 8;
    grps[8].nr = 1;
    snew(grps[8].nm_ind,1);
  }

  for(j=0; (j<ngrp); j++) {
    do_int (grps[j].nr);
    if (bRead)
      snew(grps[j].nm_ind,grps[j].nr);
    ndo_int(grps[j].nm_ind,grps[j].nr,bDum);
  }
}
