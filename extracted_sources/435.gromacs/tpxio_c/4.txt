static void do_ilist(t_ilist *ilist,bool bRead,char *name)
{
  int i;
  bool bDum=TRUE;
  
  if (!bRead)
    set_comment(name);
  ndo_int(ilist->multinr,MAXNODES,bDum);
  do_int (ilist->nr);
  if (bRead)
    snew(ilist->iatoms,ilist->nr);
  ndo_int(ilist->iatoms,ilist->nr,bDum);
  if (!bRead)
    unset_comment();
}
