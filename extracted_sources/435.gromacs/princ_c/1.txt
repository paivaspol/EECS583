static void ptrans(char *s,real **inten,real d[],real e[])
{
  int  m;
  real n,x,y,z;
  for(m=1; (m<NDIM); m++) {
    x=inten[m][1];
    y=inten[m][2];
    z=inten[m][3];
    n=x*x+y*y+z*z;
    fprintf(stderr,"%8s %8.3f %8.3f %8.3f, norm:%8.3f, d:%8.3f, e:%8.3f\n",
	    s,x,y,z,sqrt(n),d[m],e[m]);
  }
  fprintf(stderr,"\n");
}
