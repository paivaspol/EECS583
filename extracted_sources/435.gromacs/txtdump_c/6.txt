void pr_ivec(FILE *fp,int indent,char *title,int vec[],int n)
{
  int i;

  if (available(fp,vec,title))
    {
      indent=pr_title_n(fp,indent,title,n);
      for (i=0; i<n; i++)
        {
          (void) pr_indent(fp,indent);
          (void) fprintf(fp,"%s[%d]=%d\n",title,bShowNumbers?i:-1,vec[i]);
        }
    }
}
