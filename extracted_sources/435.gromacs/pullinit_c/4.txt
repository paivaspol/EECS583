static void print_whole_index(char **grpnames, atom_id **index, int *ngx, 
			      int ngrps) 
{
  int i,j;
  FILE *tmp;
  
  tmp = ffopen("indexcheck","w");
  for (i=0;i<ngrps;i++) {
    fprintf(tmp,"\nGrp %d: %s. %d elements\n",i,grpnames[i],ngx[i]);
    for (j=0;j<ngx[i];j++)
      fprintf(tmp," %d ",index[i][j]);
  }
  fflush(tmp);
}
