static void pr_sortblock(FILE *fp,char *title,int nsb,t_sortblock sb[])
{
  int i;
  
    if(fp)
    {
        fprintf(fp,"%s\n",title);
        for(i=0; (i<nsb); i++)
            fprintf(fp,"i: %5d, iatom: (%5d %5d %5d), blocknr: %5d\n",
                    i,sb[i].iatom[0],sb[i].iatom[1],sb[i].iatom[2],
                    sb[i].blocknr);
    }
}
