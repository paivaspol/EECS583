void write_gms(FILE *fp,int natoms,rvec x[],matrix box)
{
  int i,j;

  n=0;
  for(i=0;(i<natoms);i++)
    for(j=0;(j<3);j++) {
      fprintf(fp,"%8.3f",x[i][j]);
      n++;
      if (n==FPL) {
	fprintf(fp,"\n");
	n=0;
      }
    }
  if (n != 0) 
    fprintf(fp,"\n");
  if (box != NULL)
    fprintf(fp,"%8.3f%8.3f%8.3f\n",box[XX][XX],box[YY][YY],box[ZZ][ZZ]);
}
