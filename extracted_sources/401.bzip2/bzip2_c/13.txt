static 
void panic ( Char* s )
{
   fprintf ( stderr,
             "\n%s: PANIC -- internal consistency error:\n"
             "\t%s\n"
#if defined(SPEC_CPU)
             "\tThis is probably a BUG, but it may be in your COMPILER.  Please do not bother\n"
             "\tthe original author.\n",
#else
             "\tThis is a BUG.  Please report it to me at:\n"
             "\tjseward@bzip.org\n",
#endif
             progName, s );
   showFileNames();
   cleanUpAndFail( 3 );
}
