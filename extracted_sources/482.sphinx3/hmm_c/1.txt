void hmm_clear (hmm_t *h, int32 n_state)
{
    int32 i;

    h->in.score = S3_LOGPROB_ZERO;
    h->in.history = -1;
    for (i = 0; i < n_state; i++) {
	h->state[i].score = S3_LOGPROB_ZERO;
	h->state[i].history = -1;
    }
    h->out.score = S3_LOGPROB_ZERO;
    h->out.history = -1;
    
    h->bestscore = S3_LOGPROB_ZERO;
}
