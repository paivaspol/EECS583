void
kb_freehyps(kb_t *kb)
{
  int i = 0;

  if (kb->hyp_segs) {
    for (; i < kb->hyp_seglen; i++) {
      if (kb->hyp_segs[i]) {
	ckd_free(kb->hyp_segs[i]);
      }
    }
    ckd_free(kb->hyp_segs);
    kb->hyp_seglen = 0;
  }
	
  if (kb->hyp_str) {
    ckd_free(kb->hyp_str);
    kb->hyp_strlen = 0;
  }
}
