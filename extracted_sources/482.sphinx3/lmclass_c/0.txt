void lmclass_dump (lmclass_t cl, FILE *fp)
{
    lmclass_word_t w;
    
    assert (cl != NULL);
    
    fprintf (fp, "LMCLASS %s\n", cl->name);
    for (w = cl->wordlist; w; w = w->next)
	fprintf (fp, "    %s\t%d\n", w->word, w->LOGprob);
    fprintf (fp, "END %s\n", cl->name);
    
    fflush (fp);
}
