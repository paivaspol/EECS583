void feat_print (feat_t *fcb, float32 ***feat, int32 nfr, FILE *fp)
{
    int32 i, j, k;
    
    for (i = 0; i < nfr; i++) {
	fprintf (fp, "%8d:", i);

	for (j = 0; j < feat_n_stream(fcb); j++) {
	    fprintf (fp, "\t%2d:", j);

	    for (k = 0; k < feat_stream_len(fcb, j); k++)
		fprintf (fp, " %8.4f", feat[i][j][k]);
	    fprintf (fp, "\n");
	}
    }
    
    fflush (fp);
}
