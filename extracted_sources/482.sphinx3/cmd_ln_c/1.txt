static int32 arg_strlen (arg_t *defn, int32 *namelen, int32 *deflen)
{
    int32 i, l;
    
    *namelen = *deflen = 0;
    for (i = 0; defn[i].name; i++) {
	l = strlen (defn[i].name);
	if (*namelen < l)
	    *namelen = l;
	
	if (defn[i].deflt) {
	    l = strlen (defn[i].deflt);
	    if (*deflen < l)
		*deflen = l;
	}
    }

    return i;
}
