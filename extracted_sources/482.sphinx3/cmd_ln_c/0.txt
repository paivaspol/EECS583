static const char *arg_type2str (argtype_t t)
{
    switch (t) {
    case ARG_INT32:
    case REQARG_int32:
	return ("int32");
	break;
    case ARG_FLOAT32:
    case REQARG_FLOAT32:
	return ("float32");
	break;
    case ARG_FLOAT64:
    case REQARG_FLOAT64:
	return ("float64");
	break;
    case ARG_STRING:
    case REQARG_STRING:
	return ("string");
	break;
    default: E_FATAL("Unknown argument type: %d\n", t);
    }
}
