int  Pawn(int sq,int c) {
  int s = 0;
  int xside = c^1;
  
  s += 100;

  s += std_p_tropism[distance[sq][king_locs[xside]]];
  s += std_own_p_tropism[distance[sq][king_locs[c]]];

  /* pawn duo */
  if (piecet(sq+1) == pawn
      && pieceside(sq+1) == c)
      s += 5;

  /* pawn chain */
  if(piecet(sq+11) == pawn
     && pieceside(sq+11) == c)
     s += 1;
  else if (piecet(sq+13) == pawn
           && pieceside(sq+13) == c)
     s += 1;
  
  if( c == BLACK )
    s = -s;
  return s;
}
