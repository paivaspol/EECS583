void stringize_pv (char str[])
{
  char output[STR_BUFF];
  int i;

  memset(str, 0, STR_BUFF);

   for (i = 1; i < pv_length[1]; i++)
   {
        	comp_to_san (pv[1][i], output);
          	make(&pv[1][i], 0);
          	strcat (str, output);
	  	strcat (str, " ");
   }

   hash_extract_pv(7, str);

   for (i = (pv_length[1]-1); i > 0; i--)
   {
   	unmake(&pv[1][i], 0);
   }

}
