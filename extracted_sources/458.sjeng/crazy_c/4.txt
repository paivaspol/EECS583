void addHolding(int what, int who)
{

  if (Variant == Crazyhouse)
    {

      holding[who][what]++;
      
      num_holding[who]++;
      
      HHash(what, holding[who][what]);

    };

  if (who == WHITE)
    white_hand_eval += hand_value[what];
  else
    black_hand_eval += hand_value[what];

  Material += material[what];

  return;
    
}
