void
ReverseIntStack(struct intstack_s *stack)
{
  struct intstack_s *old;
  struct intstack_s *new;

  old        = stack->nxt;
  stack->nxt = NULL;
  while (old != NULL)
    {
      new        = old;		/* remove one from top of old stack */
      old        = old->nxt;
      new->nxt   = stack->nxt;  /* push it onto new stack */
      stack->nxt = new;
    }
}
