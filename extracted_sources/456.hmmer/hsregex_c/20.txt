int
main(int argc, char **argv)
{
  char *pat;
  int   ntok;
  char *s;
  int   status;

  pat  = argv[1];
  ntok = atoi(argv[2]);
  s    = argv[3];

  status = Strparse(pat, s, ntok);
  if (status == 0) {
    printf("no match\n");
  } else {
    int i;
    printf("MATCH.\n");
    for (i = 1; i <= ntok; i++) 
      printf("matched token %1d:  %s\n", i, sqd_parse[i]);
  }
}
