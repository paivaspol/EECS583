static void
writehash(Hashseq   hashseq)
{
  int idx;
  int sym;

  if (hashseq/16)
    writehash(hashseq/16);
  
  sym = (int) (hashseq % 16);
  if (sym == 0)
    putchar('-');
  else
    {
      for (idx = 0; sym != iupac[idx].code && idx < IUPACSYMNUM; idx++);
      if (idx > IUPACSYMNUM)
        printf("(%d)", sym);
      else
        putchar(iupac[idx].sym);
    }
}
