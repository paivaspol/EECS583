void
SqdBanner(FILE *fp, char *banner)
{
  fprintf(fp, "%s\n", banner);
  fprintf(fp, "SQUID %s (%s)\n", SQUID_VERSION, SQUID_DATE);
  fprintf(fp, "%s\n", SQUID_COPYRIGHT);
  fprintf(fp, "%s\n", SQUID_LICENSE);
  fprintf(fp, "- - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - - -\n");
}
