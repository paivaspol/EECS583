static void
record_conflicts (allocno_vec, len)
     int *allocno_vec;
     int len;
{
  int num;
  int ialloc_prod;

  while (--len >= 0)
    {
      num = allocno_vec[len];
      ialloc_prod = num * allocno_row_words;
      IOR_HARD_REG_SET (allocno[num].hard_reg_conflicts, hard_regs_live);
    }
}
