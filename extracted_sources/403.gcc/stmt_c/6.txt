static void
mark_goto_fixup (g)
     struct goto_fixup *g;
{
  while (g)
    {
      ggc_mark (g);
      ggc_mark_rtx (g->before_jump);
      ggc_mark_tree (g->target);
      ggc_mark_tree (g->context);
      ggc_mark_rtx (g->target_rtl);
      ggc_mark_rtx (g->stack_level);
      ggc_mark_tree (g->cleanup_list_list);

      g = g->next;
    }
}
