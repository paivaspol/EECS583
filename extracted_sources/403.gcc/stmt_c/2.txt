static void
mark_loop_nesting (n)
     struct nesting *n;
{

  while (n)
    {
      ggc_mark_rtx (n->exit_label);
      ggc_mark_rtx (n->data.loop.start_label);
      ggc_mark_rtx (n->data.loop.end_label);
      ggc_mark_rtx (n->data.loop.alt_end_label);
      ggc_mark_rtx (n->data.loop.continue_label);

      n = n->next;
    }
}
