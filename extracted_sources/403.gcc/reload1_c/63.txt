static void
reload_cse_delete_noop_set (insn, value)
     rtx insn, value;
{
  bool purge = BLOCK_FOR_INSN (insn)->end == insn;
  if (value)
    {
      PATTERN (insn) = gen_rtx_USE (VOIDmode, value);
      INSN_CODE (insn) = -1;
      REG_NOTES (insn) = NULL_RTX;
    }
  else
    delete_insn (insn);
  if (purge)
    purge_dead_edges (BLOCK_FOR_INSN (insn));
}
