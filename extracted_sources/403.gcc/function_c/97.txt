static void
emit_return_into_block (bb, line_note)
     basic_block bb;
     rtx line_note;
{
  rtx p, end;

  p = NEXT_INSN (bb->end);
  end = emit_jump_insn_after (gen_return (), bb->end);
  if (line_note)
    emit_line_note_after (NOTE_SOURCE_FILE (line_note),
			  NOTE_LINE_NUMBER (line_note), PREV_INSN (bb->end));
}
