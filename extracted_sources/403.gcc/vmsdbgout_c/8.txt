static int
write_debug_delta4 (label1, label2, comment, dosizeonly)
     char *label1;
     char *label2;
     const char *comment;
     int dosizeonly;
{
  if (!dosizeonly)
    {
      ASM_OUTPUT_DEBUG_DELTA4 (asm_out_file, label1, label2);
      if (flag_verbose_asm)
	fprintf (asm_out_file, "\t%s %s", ASM_COMMENT_START, comment);
      fputc ('\n', asm_out_file);
    }

  return 4;
}
