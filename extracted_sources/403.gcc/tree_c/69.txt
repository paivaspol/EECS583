static tree
make_vector (mode, innertype, unsignedp)
     enum machine_mode mode;
     tree innertype;
     int unsignedp;
{
  tree t;

  t = make_node (VECTOR_TYPE);
  TREE_TYPE (t) = innertype;
  TYPE_MODE (t) = mode;
  TREE_UNSIGNED (TREE_TYPE (t)) = unsignedp;
  finish_vector_type (t);

  return t;
}
