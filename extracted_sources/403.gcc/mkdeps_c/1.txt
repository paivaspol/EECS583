void
deps_free (d)
     struct deps *d;
{
  unsigned int i;

  if (d->targetv)
    {
      for (i = 0; i < d->ntargets; i++)
	free ((PTR) d->targetv[i]);
      free (d->targetv);
    }

  if (d->depv)
    {
      for (i = 0; i < d->ndeps; i++)
	free ((PTR) d->depv[i]);
      free (d->depv);
    }

  free (d);
}
