static int
compare_blocks (v1, v2)
     const PTR v1;
     const PTR v2;
{
  tree b1 = *((const tree *) v1);
  tree b2 = *((const tree *) v2);
  char *p1 = (char *) BLOCK_ABSTRACT_ORIGIN (b1);
  char *p2 = (char *) BLOCK_ABSTRACT_ORIGIN (b2);

  if (p1 == p2)
    return 0;
  return p1 < p2 ? -1 : 1;
}
