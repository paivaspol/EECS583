static void
df_refs_unlink (df, blocks)
     struct df *df;
     bitmap blocks;
{
  basic_block bb;

  if (blocks)
    {
      FOR_EACH_BB_IN_BITMAP (blocks, 0, bb,
      {
	df_bb_refs_unlink (df, bb);
      });
    }
  else
    {
      FOR_ALL_BBS (bb,
      {
	df_bb_refs_unlink (df, bb);
      });
    }
}
