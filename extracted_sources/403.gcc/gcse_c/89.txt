static int
pre_expr_reaches_here_p (occr_bb, expr, bb)
     basic_block occr_bb;
     struct expr *expr;
     basic_block bb;
{
  int rval;
  char *visited = (char *) xcalloc (n_basic_blocks, 1);

  rval = pre_expr_reaches_here_p_work (occr_bb, expr, bb, visited);

  free (visited);
  return rval;
}
