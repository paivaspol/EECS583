static int
expr_reaches_here_p (occr, expr, bb, check_self_loop)
     struct occr *occr;
     struct expr *expr;
     basic_block bb;
     int check_self_loop;
{
  int rval;
  char *visited = (char *) xcalloc (n_basic_blocks, 1);

  rval = expr_reaches_here_p_work (occr, expr, bb, check_self_loop, visited);
  
  free (visited);
  return rval;
}
