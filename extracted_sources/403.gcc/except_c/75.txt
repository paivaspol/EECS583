bool
nothrow_function_p ()
{
  rtx insn;

  if (! flag_exceptions)
    return true;

  for (insn = get_insns (); insn; insn = NEXT_INSN (insn))
    if (can_throw_external (insn))
      return false;
  for (insn = current_function_epilogue_delay_list; insn;
       insn = XEXP (insn, 1))
    if (can_throw_external (insn))
      return false;

  return true;
}
