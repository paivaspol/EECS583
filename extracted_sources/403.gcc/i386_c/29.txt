int
ix86_function_arg_boundary (mode, type)
     enum machine_mode mode;
     tree type;
{
  int align;
  if (!TARGET_64BIT)
    return PARM_BOUNDARY;
  if (type)
    align = TYPE_ALIGN (type);
  else
    align = GET_MODE_ALIGNMENT (mode);
  if (align < PARM_BOUNDARY)
    align = PARM_BOUNDARY;
  if (align > 128)
    align = 128;
  return align;
}
