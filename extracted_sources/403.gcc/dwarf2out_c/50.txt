{

      return "DW_FORM_addr";
    case DW_FORM_block2:

    case DW_FORM_block4:
      return "DW_FORM_block4";

    case DW_FORM_strp:
      return "DW_FORM_strp";
    case DW_FORM_udata:
      return "DW_FORM_udata";
    case DW_FORM_ref_addr:
      return "DW_FORM_ref_addr";

      return "DW_FORM_ref1";
    case DW_FORM_ref2:
